<script>
    // --- PEGA TU ENLACE DE GOOGLE ABAJO ENTRE LAS COMILLAS ---
    const SCRIPT_URL = "https://script.google.com/macros/s/AKfycbzV5AbZynKHswQ3f-_87_rOrcLuD8ghaexkzEzyLqVoQ9AuJaiOesEuHjzEXebkiBDufw/exec";
    
    const planesDirectv = ["Prepago HD", "Oro Plus", "Platino", "DIRECTV GO"];
    const planesClaro = ["Triple Play", "Doble Play", "Solo Internet", "Claro TV Superior", "Pospago M√≥vil"];
    let ventasGlobales = [];

    document.addEventListener('DOMContentLoaded', () => {
        actualizarPlanes();
        fetchData();
    });

    function actualizarPlanes() {
        const empresa = document.getElementById('empresa');
        const planSelect = document.getElementById('plan');
        const lista = empresa.value === 'DIRECTV' ? planesDirectv : planesClaro;
        empresa.className = empresa.value === 'DIRECTV' ? 'bg-directv' : 'bg-claro';
        planSelect.innerHTML = "";
        lista.forEach(p => planSelect.innerHTML += `<option value="${p}">${p}</option>`);
    }

    async function fetchData() {
        document.getElementById('loadingStatus').style.display = 'block';
        try {
            const response = await fetch(SCRIPT_URL + "?action=get");
            const data = await response.json();
            // Invertir el orden para que la √∫ltima venta salga primero
            ventasGlobales = data.reverse(); 
            render(ventasGlobales);
        } catch (e) { 
            console.error("Error cargando datos", e); 
        }
        document.getElementById('loadingStatus').style.display = 'none';
    }

    async function guardarVenta() {
        const btn = document.getElementById('btnGuardar');
        const data = {
            action: "add",
            id: Date.now(),
            empresa: document.getElementById('empresa').value,
            cedula: document.getElementById('cedula').value.trim(),
            nombre: document.getElementById('nombre').value.trim(),
            tel: document.getElementById('tel').value.trim(),
            correo: document.getElementById('correo').value.trim(),
            deco: document.getElementById('deco').value.trim().toUpperCase(),
            plan: document.getElementById('plan').value,
            fechaReg: new Date().toLocaleString(),
            estado: 'Pendiente',
            fechaEntrega: ''
        };

        if(!data.cedula || !data.nombre || !data.deco) {
            alert("‚ö†Ô∏è Por favor completa los datos b√°sicos (C√©dula, Nombre y Equipo)");
            return;
        }

        btn.disabled = true;
        btn.innerText = "Sincronizando con la nube...";

        try {
            await fetch(SCRIPT_URL, { 
                method: 'POST', 
                mode: 'no-cors', // Importante para evitar bloqueos de navegador
                body: JSON.stringify(data) 
            });
            
            // Como usamos no-cors, esperamos un segundo y refrescamos
            setTimeout(async () => {
                limpiarCampos();
                await fetchData();
                btn.disabled = false;
                btn.innerText = "REGISTRAR Y SINCRONIZAR";
                alert("‚úÖ Venta registrada correctamente en la nube.");
            }, 1500);

        } catch (e) { 
            alert("‚ùå Error de conexi√≥n. Revisa tu internet."); 
            btn.disabled = false;
            btn.innerText = "REGISTRAR Y SINCRONIZAR";
        }
    }

    function render(lista) {
        const body = document.querySelector('#mainTable tbody');
        body.innerHTML = "";
        if(lista.length === 0) {
            body.innerHTML = "<tr><td colspan='6' style='text-align:center'>No hay registros a√∫n.</td></tr>";
            return;
        }
        lista.forEach(v => {
            const esOk = v.estado === 'Entregado';
            body.innerHTML += `
                <tr>
                    <td><small>${v.fechareg || v.fechaReg}</small></td>
                    <td><span class="badge-marca" style="background:${v.empresa === 'DIRECTV' ? 'var(--directv)' : 'var(--claro)'}">${v.empresa}</span></td>
                    <td><b>${v.nombre}</b><br><small>CC: ${v.cedula} | üìû ${v.tel}</small></td>
                    <td><span class="badge-deco">${v.deco}</span></td>
                    <td>${v.plan}</td>
                    <td>
                        <span class="${esOk ? 'status-entregado' : 'status-pendiente'}">${v.estado}</span>
                        ${esOk ? `<br><small>‚úÖ ${v.fechaentrega || v.fechaEntrega}</small>` : ''}
                    </td>
                </tr>
            `;
        });
    }

    function filtrar() {
        const val = document.getElementById('buscarInput').value.toLowerCase();
        const filtradas = ventasGlobales.filter(v => 
            v.nombre.toLowerCase().includes(val) || 
            v.cedula.toString().includes(val) || 
            v.deco.toLowerCase().includes(val) ||
            v.empresa.toLowerCase().includes(val)
        );
        render(filtradas);
    }

    function limpiarCampos() {
        ['cedula','nombre','tel','correo','deco'].forEach(id => document.getElementById(id).value = "");
    }
